Código de la Solución

function memoize(fn) {
    let c = new Map(), o = new WeakMap(), i = 0;
    return function(...a) {
        let k = a.map(arg => 
            arg === null ? 'null' :
            arg === undefined ? 'undefined' :
            typeof arg === 'object' ? `obj:${o.has(arg) ? o.get(arg) : o.set(arg, ++i).get(arg)}` :
            `${typeof arg}:${arg}`
        ).join('|');
        return c.has(k) ? c.get(k) : c.set(k, fn(...a)).get(k);
    };
}
